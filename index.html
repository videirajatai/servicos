Substitua esta URL pela URL gerada no seu Google Apps Script! const API_URL = 'COLE_A_URL_DO_SEU_WEB_APP_AQUI'; // Função para salvar o orçamento na planilha async function salvarOrcamento() { const cliente = document.getElementById('nomeCliente').value; const btn = document.querySelector('#view-orcamento .btn-primary'); if (!cliente) { alert("Por favor, preencha o nome do cliente."); return; } // Muda o visual do botão para mostrar que está carregando btn.innerText = 'Salvando...'; btn.disabled = true; // Coleta todos os serviços digitados const servicos = []; let total = 0; document.querySelectorAll('.servico-item').forEach(item => { const desc = item.querySelector('.desc-servico').value; const qtd = parseFloat(item.querySelector('.qtd-servico').value) || 0; const valor = parseFloat(item.querySelector('.valor-servico').value) || 0; if (desc) { // Só adiciona se tiver descrição servicos.push({ descricao: desc, quantidade: qtd, valorUnitario: valor }); total += (qtd * valor); } }); const payload = { action: 'salvarOrcamento', cliente: cliente, servicos: servicos, total: total }; try { // Envia os dados para o Google Apps Script const response = await fetch(API_URL, { method: 'POST', body: JSON.stringify(payload) }); const resultado = await response.json(); if (resultado.status === 'sucesso') { alert('Orçamento salvo com sucesso na aba "Orcamentos"!'); // Opcional: Aqui podemos limpar o formulário para o próximo } else { alert('Erro no servidor: ' + resultado.mensagem); } } catch (erro) { alert('Erro de conexão. Verifique se a URL da API está correta.'); console.error(erro); } finally { // Restaura o botão btn.innerText = 'Salvar e Gerar Resumo'; btn.disabled = false; } } // Função para salvar entradas e saídas financeiras async function salvarLancamento() { const dataLanc = document.getElementById('finData').value; const tipo = document.getElementById('finTipo').value; const desc = document.getElementById('finDesc').value; const valor = document.getElementById('finValor').value; const btn = document.querySelector('#view-financeiro .btn-success'); if (!dataLanc || !desc || !valor) { alert('Preencha data, descrição e valor para registrar.'); return; } btn.innerText = 'Registrando...'; btn.disabled = true; const payload = { action: 'salvarFinanceiro', data: dataLanc, tipo: tipo, descricao: desc, valor: parseFloat(valor) }; try { const response = await fetch(API_URL, { method: 'POST', body: JSON.stringify(payload) }); const resultado = await response.json(); if (resultado.status === 'sucesso') { alert('Lançamento salvo na aba "Financeiro"!'); document.getElementById('finDesc').value = ''; // Limpa a descrição document.getElementById('finValor').value = ''; // Limpa o valor } else { alert('Erro no servidor: ' + resultado.mensagem); } } catch (erro) { alert('Erro de conexão.'); } finally { btn.innerText = 'Registrar'; btn.disabled = false; } }
